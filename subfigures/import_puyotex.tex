% In order to release the PuyoTeX package in a flat, self-contained directory structure
% this file modifies the TeX and Python paths to find puyotex.sty and puyotex.py.
% Rather than \usepackage{puyotex}, TeX files in this subdirectory need to
% \input{import_puyotex.tex} instead. A LaTeX warning is expected: "You have requested
% package '../puyotex' but the package provides 'puyotex'."
%
% These path modifications are not necessary when the package is installed in the
% appropriate TeX library filesystem (MikTeX or other).

\usepackage{pythontex}

\begin{pycode}
import sys, pathlib
sys.path.append(str(pathlib.Path(__file__).parents[2]))
\end{pycode}

\usepackage{../puyotex}
